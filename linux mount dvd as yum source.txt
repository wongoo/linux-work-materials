
----------------------------------------------------------------
###挂载ISO
----------------------------------------------------------------
head /dev/dvd | file -
	/dev/stdin: ISO 9660 CD-ROM filesystem data 'NOV_25_2007   
mkdir /mnt/dvd/
mount /dev/dvd /mnt/dvd/

mount -o loop /tmp/Asianux-4.4-rtm-x86_64-dvddisc-201408011123.iso /mnt/dvd/

----------------------------------------------------------------
###创建新的yum源指向挂载的ISO目录
----------------------------------------------------------------
vi /etc/yum.repos.d/iso.repo

[iso]
name=iso
baseurl=file:///mnt/dvd
enabled=1
gpgcheck=0

----------------------------------------------------------------
###备份系统默认的yum源
----------------------------------------------------------------
再把 rhel-source.repo 改名:

cd /etc/yum.repos.d
mv rhel-source.repo rhel-source.repo.bak

----------------------------------------------------------------
###刷新yum缓存
----------------------------------------------------------------
运行yum makecache生成缓存
yum makecache

----------------------------------------------------------------
###测试
----------------------------------------------------------------
yum search httpd
